Une collection de fichiers json

Nous obtenons parfois (souvent) en rponse une requte une collection de fichiers json, Python nous permet l aussi au prix de quelques lignes de codes de disposer d'un matriau exploitable

import os


# Initialiser une liste pour stocker les donnes extraites
rows = []

# Spcifier le chemin du rpertoire contenant les fichiers JSON
directory = 'extrad'

# Itrer sur tous les fichiers dans le rpertoire et sous-rpertoires
# nous ouvrons un  un chaque fichier json prsent dans le dossier
for root, dirs, files in os.walk(directory):
    for filename in files:
        if filename.endswith('.json'):
            filepath = os.path.join(root, filename)

            # Charger les donnes JSON du fichier
            with open(filepath, 'r') as file:
                data = json.load(file)

            # Extraire les informations de chaque document
            for document in data['documents']:
                publication_date = document.get('publication_date')
                publication_title = document.get('publication_title')
                
                for page in document.get('pages', []):
                    for block in page.get('blocks', []):
                        text = block.get('text', '')
                        if "extrad" in text.lower():
                            row = {
                                "publication_date": publication_date,
                                "publication_title": publication_title,
                                "texte": text
                            }
                            rows.append(row)

# Crer le DataFrame  partir des donnes extraites
df = pd.DataFrame(rows)

# Enregistrer le DataFrame dans un fichier CSV
df.to_csv('test.csv', index=False)
